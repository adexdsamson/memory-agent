  2. buffer.push(turn, session_id)   (in-memory, no await)
  3. if looks_like_durable_claim(content, type_hint, durable):
       embeddings = await embedder.embed([content])   # ONE call
       record = MemoryRecord(user_id=user_id, session_id=session_id, content=content,
                             summary=content[:80].strip(), provisional=True, t0_ref=t0_ref,
                             embedding_model=embedder.__class__.__name__,
                             embedding_dim=embedder.dim, embedding_version=getattr(embedder, 'version', None),
                             record_type=resolved_record_type, ...)
       await record_store.upsert(record)
       await vector_index.upsert_vector(record.id, embeddings[0])
  4. await staging_queue.put({turn, t0_ref})   # asyncio.Queue
  return t0_ref (and record.id if provisional was written)

Recall path logic (from mnema-build-plan.md §5, Phase 1 subset — no BM25/graph/RRF):
  1. q_vec = (await embedder.embed([query]))[0]
  2. dense_hits = await vector_index.vector_search(q_vec, k=30, user_id=user_id, agent_id=agent_id)
     # returns list[tuple[str, float]] — (record_id, distance)
  3. buffer_candidates = buffer.as_candidates(session_id=None)  # all sessions for user in buffer
  4. all_ids = dedupe: dense_hit ids + buffer turn ids, buffer-wins on duplicates
  5. records = [await record_store.get(id) for id in all_ids if id is not None]
     # For buffer turns that don't have a T1 record yet, include the Turn content directly
  6. await record_store.update(rec.id, access_count=rec.access_count+1, last_accessed=datetime.utcnow())
     for each record from T1 (not buffer-only turns)
  7. return records (list[MemoryRecord])
