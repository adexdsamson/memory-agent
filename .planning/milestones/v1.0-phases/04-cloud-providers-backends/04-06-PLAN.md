    Class OSSS3Store (no Protocol inheritance):
      __init__(self, bucket: str, *, aws_access_key_id: str, aws_secret_access_key: str, endpoint_url: str | None = None, region_name: str = "us-east-1") -> None:
        self._bucket = bucket
        self._client = boto3.client("s3", aws_access_key_id=aws_access_key_id, aws_secret_access_key=aws_secret_access_key, endpoint_url=endpoint_url, region_name=region_name, config=Config(s3={"addressing_style": "path"}))

      async append(self, session_id: str, turn: Turn) -> str:
        _validate_session_id(session_id)
        def _call() -> str:
          resp = self._client.list_objects_v2(Bucket=self._bucket, Prefix=f"{session_id}/")
          offset = resp.get("KeyCount", 0)
          key = f"{session_id}/{offset}.json"
          self._client.put_object(Bucket=self._bucket, Key=key, Body=turn.model_dump_json().encode())
          return f"t0://{session_id}/{offset}"
        return await asyncio.to_thread(_call)

      async get(self, ref: str) -> Turn:
        if not ref.startswith("t0://"):
          raise ValueError(f"Invalid t0 ref: {ref!r}")
        _, _, path = ref.partition("t0://")
        parts = path.split("/", 1)
        if len(parts) != 2:
          raise ValueError(f"Malformed t0 ref: {ref!r}")
        session_id, offset_str = parts[0], parts[1]
        _validate_session_id(session_id)
        def _call() -> Turn:
          key = f"{session_id}/{offset_str}.json"
          obj = self._client.get_object(Bucket=self._bucket, Key=key)
          return Turn.model_validate(json.loads(obj["Body"].read()))
        return await asyncio.to_thread(_call)

      async archive(self, record: MemoryRecord) -> str:
        def _call() -> str:
          key = f"archived/{record.id}.json"
          self._client.put_object(Bucket=self._bucket, Key=key, Body=record.model_dump_json().encode())
          return f"archived://{record.id}"
        return await asyncio.to_thread(_call)

      async append_audit(self, entry: dict) -> None:
        def _call() -> None:
          ts = entry.get("evicted_at", "0").replace(":", "-").replace(" ", "_")
          rid = entry.get("record_id", "unknown")
          key = f"eviction_audit/{ts}_{rid}.json"
          self._client.put_object(Bucket=self._bucket, Key=key, Body=json.dumps(entry).encode())
        await asyncio.to_thread(_call)

    Note: append() is not atomic under concurrent writes (list_objects_v2 + put_object). Document this in the class docstring as acceptable for v1 single-writer usage. The one-object-per-turn design is idempotent on retry (same offset key is overwritten, not duplicated).
