      async def evict(self, *, user_id: str) -> int:
          "Run a batch decay-based eviction pass for user_id. Returns count of evicted records."
          from mnema.core.decay import decay_pass, LAMBDA_DECAY as _  # noqa: PLC0415, F401
          from datetime import datetime, timezone  # noqa: PLC0415
          now = datetime.now(timezone.utc)
          evicted = 0
          async for record, score in decay_pass(self._t1, user_id, now=now):
              if score >= KEEP_THRESHOLD:
                  continue
              # No not-protected guard — decay_pass structural guarantee (FORG-03)
              # Comment: "protected records cannot reach this point; proven by Hypothesis test"

              await self._t0.append_audit(entry)
              evicted += 1
          return evicted
