      def _drain_queue(self) -> list[dict[str, Any]]:
        """Drain asyncio.Queue using get_nowait() + QueueEmpty catch (RESEARCH.md §Code Examples)."""
        items: list[dict[str, Any]] = []
        while True:
            try:
                items.append(self._staging_queue.get_nowait())
            except asyncio.QueueEmpty:
                break
        return items

      async def _process_turn(self, content: str, t0_ref: str, user_id: str, session_id: str) -> None:
        """Extract records from one turn and reconcile/resolve each."""
        # Step 2: LLM extraction
        extraction_prompt = f"{_EXTRACT_SENTINEL} {content}"
        raw_response = await self._llm.complete(extraction_prompt)
        try:
            extracted_list: list[dict[str, Any]] = json.loads(raw_response)
        except (json.JSONDecodeError, ValueError):
            return  # Malformed LLM response — skip, log implicitly via return

        for ext in extracted_list:
            extracted_content = str(ext.get("content", content))
            # Step 3: Safety pin pass — content rule overrides LLM salience (D2-03)
            if _is_safety_claim(extracted_content):
                ext["protected"] = True
                ext["salience"] = 1.0
                ext["record_type"] = "fact"
            else:
                ext.setdefault("protected", False)
                ext.setdefault("salience", 0.5)
                ext.setdefault("record_type", "preference")

            # Step 4: Provisional reconcile-by-t0_ref (CONS-06)
            existing_provisional = await self._record_store.find_by_t0_ref(t0_ref, user_id)
            if existing_provisional is not None:
                # Upgrade in place — never delete-and-reinsert (D2-12)
                await self._record_store.update(
                    existing_provisional.id,
                    provisional=False,
                    record_type=ext["record_type"],
                    salience=float(ext.get("salience", 0.5)),
                    summary=str(ext.get("summary", extracted_content[:60])),
                    keywords=ext.get("keywords", []),
                    protected=bool(ext.get("protected", False)),
                )
                # Update protected status per content-driven rule (never clear protected) —
                # if existing_provisional.protected is already True, must stay True
                if existing_provisional.protected and not ext.get("protected"):
                    await self._record_store.update(existing_provisional.id, protected=True)
                continue  # No entity resolution needed for reconciled provisionals

            # Step 5: Entity resolution (new records only — D2-04: no new embed for reconciled)
            embeddings = await self._embedder.embed([extracted_content])
            embedding = embeddings[0]
            record_type_enum = RecordType(ext["record_type"]) if ext["record_type"] in RecordType._value2member_map_ else RecordType.PREFERENCE
            candidates = await self._vector_index.vector_search(
                embedding, k=5, user_id=user_id
            )
            near_match: Optional[MemoryRecord] = None
            for cand_id, dist in candidates:
                if dist <= ENTITY_MAX_DISTANCE:
                    cand = await self._record_store.get(cand_id)
                    if cand is not None and cand.record_type == record_type_enum:
                        near_match = cand
                        break

            if near_match is None:
                # No match → insert new confirmed record
                await self._insert_new_confirmed(ext, extracted_content, embedding,
                                                  user_id, session_id, t0_ref)
            else:
                # Step 6: Contradiction judge
                judge_prompt = f"{_JUDGE_SENTINEL} {near_match.content}\n{extracted_content}"
                verdict = (await self._llm.complete(judge_prompt)).strip().lower()
                await self._apply_verdict(verdict, near_match, ext, extracted_content,
                                           embedding, user_id, session_id, t0_ref)

      async def _apply_verdict(self, verdict: str, existing: MemoryRecord,
                                ext: dict[str, Any], new_content: str,
                                embedding: list[float], user_id: str,
                                session_id: str, t0_ref: str) -> None:
        """Apply contradiction/refine/distinct verdict with CONS-08 gate."""
        if verdict == "refine":
            # CONS-05: non-contradicting refinement — merge in place
            await self._record_store.update(
                existing.id,
                content=new_content,
                summary=new_content[:60].strip(),
                salience=float(ext.get("salience", existing.salience)),
                keywords=ext.get("keywords", existing.keywords),
            )
            return

        if verdict == "contradict":
            # CONS-08: structural gate — protected/FACT records are NEVER auto-superseded
            if existing.protected or existing.record_type == RecordType.FACT:
                # Record contradiction_pending edge on the EXISTING record (audit trail)
                new_edge = {
                    "rel": "contradiction_pending",
                    "target": f"unresolved_{t0_ref}",
                    "ts": datetime.now(timezone.utc).isoformat(),
                }
                new_edges = list(existing.graph_edges) + [new_edge]  # Pitfall 5: new list
                await self._record_store.update(existing.id, graph_edges=new_edges)
                return  # Leave existing live — no supersession

            # Not protected/FACT → atomic supersession (CONS-04)
            new_edge_supersedes = {"rel": "supersedes", "target": existing.id}
            new_record = MemoryRecord(
                user_id=user_id,
                session_id=session_id,
                record_type=RecordType(ext.get("record_type", "preference")),
                content=new_content,
                summary=new_content[:60].strip(),
                keywords=ext.get("keywords", []),
                salience=float(ext.get("salience", 0.5)),
                protected=bool(ext.get("protected", False)),
                provisional=False,
                t0_ref=t0_ref,
                graph_edges=[new_edge_supersedes],  # new list per Pitfall 5
                embedding_model=self._embedder.__class__.__name__,
                embedding_dim=self._embedder.dim,
                embedding_version=getattr(self._embedder, "version", None),
            )
            await self._record_store.supersede(existing.id, new_record, embedding)
            return

        # verdict == "distinct" or unrecognised → insert as new confirmed record
        await self._insert_new_confirmed(ext, new_content, embedding,
                                          user_id, session_id, t0_ref)

      async def _insert_new_confirmed(self, ext: dict[str, Any], content: str,
                                       embedding: list[float], user_id: str,
                                       session_id: str, t0_ref: str) -> None:
        """Insert a new confirmed (non-provisional) record."""
        record = MemoryRecord(
            user_id=user_id,
            session_id=session_id,
            record_type=RecordType(ext.get("record_type", "preference")),
            content=content,
            summary=str(ext.get("summary", content[:60])).strip(),
            keywords=ext.get("keywords", []),
            salience=float(ext.get("salience", 0.5)),
            protected=bool(ext.get("protected", False)),
            provisional=False,
            t0_ref=t0_ref,
            embedding_model=self._embedder.__class__.__name__,
            embedding_dim=self._embedder.dim,
            embedding_version=getattr(self._embedder, "version", None),
        )
        await self._record_store.upsert(record)
        await self._vector_index.upsert_vector(record.id, embedding)

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| LLM response → json.loads | LLM (real or stub) returns a string; consolidation core parses it with try/except |
| staging queue item → user_id | user_id comes from the WritePath call site, not from LLM output |
| existing record protected flag → supersession gate | protected is a SQLite INTEGER column; the gate reads it from the retrieved MemoryRecord |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-09 | Tampering | LLM response injection — malformed JSON from complete() | mitigate | try/except json.JSONDecodeError with return (skip record). LLM cannot inject SQL — extraction only parses the response string. |
| T-02-10 | Tampering | CONS-08 gate — protected/FACT record auto-supersession | mitigate | Structural two-branch early return in _apply_verdict(): `if existing.protected or existing.record_type == RecordType.FACT: ... return`. LLM verdict never reaches supersession path for protected/FACT. |
| T-02-11 | Tampering | Consolidation re-classifies protected record to non-protected | mitigate | In provisional reconcile path: `if existing_provisional.protected and not ext.get("protected"): await update(protected=True)`. Protected flag is monotonic upward — consolidation can SET but never CLEAR it. |
| T-02-12 | Tampering | Supersession atomicity — partial write | mitigate | Delegated to SqliteT1.supersede() (Plan 03) which wraps all three statements in try/except rollback. ConsolidationPipeline never calls the three statements separately. |
| T-02-13 | Elevation of Privilege | Cross-user record creation | mitigate | user_id propagated from staging item (write_path.py change) to every MemoryRecord construction and every record_store call. Never read from LLM output. |
| T-02-14 | Information Disclosure | contradiction_pending edge reveals suppressed content | accept | Phase 2 is fully local, single-user. Multi-user MCP surface is Phase 3/4 — flag for review at that point. |
</threat_model>
