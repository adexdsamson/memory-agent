    async def reindex_all(t1: Any, embedder: Any, user_id: str) -> int:
      Docstring: "Re-embed all live records for user_id using the current embedder. Returns the count of records re-embedded. Preserves all record fields including protected flag — only the vector embeddings are updated."
      count = 0
      async for record in t1.live_records(user_id):

        [vec] = await embedder.embed([text])
        await t1.upsert_vector(record.id, vec)
        count += 1
      return count
