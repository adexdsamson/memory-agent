InProcessScheduler (exact analog to copy from):
  JOB_ID = "consolidate"
  class InProcessScheduler:
    JOB_ID: str = JOB_ID
    def __init__(self) -> None:
      self._scheduler: AsyncIOScheduler = AsyncIOScheduler()
    async def start(self) -> None: self._scheduler.start()
    async def schedule(self, fn: Any, *, every_seconds: int) -> None:
      self._scheduler.add_job(fn, "interval", seconds=every_seconds, id=self.JOB_ID, next_run_time=None)
    async def trigger_now(self) -> None:
      job = self._scheduler.get_job(self.JOB_ID)
      if job is not None: job.modify(next_run_time=datetime.now())
    async def shutdown(self) -> None:
      self._scheduler.shutdown(wait=False)

      JOB_ID: str = JOB_ID

        trigger = CronTrigger.from_crontab(self._cron)
